-- Bloque 1 — Extensión para búsqueda de texto

-- Bloque 3 — Función auxiliar para updated_at
CREATE OR REPLACE FUNCTION public.set_updated_at()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$;

-- Bloque 4 — Tabla usuarios
CREATE TABLE public.usuarios (
  id             uuid PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  nombre         text NOT NULL,
  email          text NOT NULL,
  rol            public.rol_type NOT NULL DEFAULT 'cliente',
  fecha_registro timestamptz DEFAULT now(),
  updated_at     timestamptz DEFAULT now(),
  eliminado      boolean NOT NULL DEFAULT false
);

CREATE TRIGGER usuarios_updated_at
  BEFORE UPDATE ON public.usuarios
  FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();

-- Bloque 5 — Trigger automático de perfil al registrarse
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
  INSERT INTO public.usuarios (id, nombre, email, rol)
  VALUES (
    NEW.id,
    COALESCE(NEW.raw_user_meta_data->>'nombre', ''),
    NEW.email,
    COALESCE((NEW.raw_user_meta_data->>'rol')::public.rol_type, 'cliente')
  )
  ON CONFLICT (id) DO NOTHING;
  RETURN NEW;
END;
$$;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

-- Bloque 7 — Tabla tiendas
CREATE TABLE public.tiendas (
  id             uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  artesano_id    uuid NOT NULL REFERENCES public.usuarios(id) ON DELETE CASCADE,
  nombre         text NOT NULL,
  descripcion    text,
  logo_url       text,
  logo_file_path text,
  estado         public.estado_tienda_type NOT NULL DEFAULT 'activa',
  fecha_creacion timestamptz DEFAULT now(),
  updated_at     timestamptz DEFAULT now()
);

CREATE INDEX idx_tiendas_artesano_id ON public.tiendas(artesano_id);
CREATE INDEX idx_tiendas_estado ON public.tiendas(estado);

CREATE TRIGGER tiendas_updated_at
  BEFORE UPDATE ON public.tiendas
  FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();

-- Bloque 8 — Tabla productos
CREATE TABLE public.productos (
  id             uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tienda_id      uuid NOT NULL REFERENCES public.tiendas(id) ON DELETE CASCADE,
  nombre         text NOT NULL,
  descripcion    text,
  precio         numeric(10,2) NOT NULL CHECK (precio >= 0),
  stock          integer NOT NULL DEFAULT 0 CHECK (stock >= 0),
  estado         public.estado_producto_type NOT NULL DEFAULT 'activo',
  metadatos      jsonb,
  fecha_creacion timestamptz DEFAULT now(),
  updated_at     timestamptz DEFAULT now()
);

CREATE INDEX idx_productos_tienda_id ON public.productos(tienda_id);
CREATE INDEX idx_productos_estado ON public.productos(estado);
CREATE INDEX idx_productos_nombre_trgm ON public.productos USING gin (nombre gin_trgm_ops);

CREATE TRIGGER productos_updated_at
  BEFORE UPDATE ON public.productos
  FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();

-- Bloque 9 — Tabla producto_categoria (relación productos ↔ categorías)
CREATE TABLE public.producto_categoria (
  producto_id  uuid NOT NULL REFERENCES public.productos(id) ON DELETE CASCADE,
  categoria_id uuid NOT NULL REFERENCES public.categorias(id) ON DELETE CASCADE,

);

CREATE INDEX idx_producto_categoria_cat ON public.producto_categoria(categoria_id);

-- Bloque 10 — Tabla imagenes_productos

-- Bloque 11 — Tabla carritos
CREATE TABLE public.carritos (

  cliente_id uuid NOT NULL REFERENCES public.usuarios(id) ON DELETE CASCADE

-- Bloque 12 — Tabla items_carrito
CREATE TABLE public.items_carrito (
  id                uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  carrito_id        uuid NOT NULL REFERENCES public.carritos(id) ON DELETE CASCADE,
  producto_id       uuid NOT NULL REFERENCES public.productos(id) ON DELETE CASCADE,
  cantidad          integer NOT NULL DEFAULT 1 CHECK (cantidad >= 1),
  precio_al_agregar numeric(10,2),

CREATE INDEX idx_items_carrito_carrito ON public.items_carrito(carrito_id);

-- Bloque 14 — Tabla pedidos
CREATE TABLE public.pedidos (
  id           uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  cliente_id   uuid NOT NULL REFERENCES public.usuarios(id),
  direccion_id uuid NOT NULL REFERENCES public.direcciones(id),

-- Bloque 15 — Tabla detalle_pedido
CREATE TABLE public.detalle_pedido (
  pedido_id   uuid NOT NULL REFERENCES public.pedidos(id) ON DELETE CASCADE,
  producto_id uuid NOT NULL REFERENCES public.productos(id),
  cantidad    integer NOT NULL CHECK (cantidad >= 1),

  PRIMARY KEY (pedido_id, producto_id)
);

CREATE INDEX idx_detalle_pedido_pedido ON public.detalle_pedido(pedido_id);

-- Bloque 16 — Tabla pagos
CREATE TABLE public.pagos (

CREATE INDEX idx_pagos_pedido ON public.pagos(pedido_id);
CREATE UNIQUE INDEX idx_pagos_stripe ON public.pagos(stripe_payment_intent_id)
  WHERE stripe_payment_intent_id IS NOT NULL;

-- Bloque 17 — Tabla reseñas
CREATE TABLE public."reseñas" (

CREATE INDEX idx_resenas_producto ON public."reseñas"(producto_id);

-- ===== USUARIOS =====
CREATE POLICY "ver propio perfil" ON public.usuarios
  FOR SELECT USING (auth.uid() = id);
CREATE POLICY "crear propio perfil" ON public.usuarios
  FOR INSERT WITH CHECK (auth.uid() = id);
CREATE POLICY "actualizar propio perfil" ON public.usuarios
  FOR UPDATE USING (auth.uid() = id);

-- ===== TIENDAS =====
CREATE POLICY "tiendas activas son públicas" ON public.tiendas
  FOR SELECT USING (estado = 'activa' OR artesano_id = auth.uid());
CREATE POLICY "artesano crea su tienda" ON public.tiendas
  FOR INSERT WITH CHECK (artesano_id = auth.uid());
CREATE POLICY "artesano edita su tienda" ON public.tiendas
  FOR UPDATE USING (artesano_id = auth.uid());

-- ===== PRODUCTOS =====
CREATE POLICY "productos activos son públicos" ON public.productos
  FOR SELECT USING (
    estado = 'activo' OR
    tienda_id IN (SELECT id FROM public.tiendas WHERE artesano_id = auth.uid())
  );
CREATE POLICY "artesano crea productos" ON public.productos
  FOR INSERT WITH CHECK (
    tienda_id IN (SELECT id FROM public.tiendas WHERE artesano_id = auth.uid())
  );
CREATE POLICY "artesano edita sus productos" ON public.productos
  FOR UPDATE USING (
    tienda_id IN (SELECT id FROM public.tiendas WHERE artesano_id = auth.uid())
  );
CREATE POLICY "artesano elimina sus productos" ON public.productos
  FOR DELETE USING (
    tienda_id IN (SELECT id FROM public.tiendas WHERE artesano_id = auth.uid())
  );

-- ===== CATEGORIAS =====

  FOR SELECT USING (true);

    producto_id IN (
      SELECT p.id FROM public.productos p
      JOIN public.tiendas t ON p.tienda_id = t.id

FASE 5 — Bucket de Storage y políticas
sql
-- Cualquiera puede ver las imágenes
CREATE POLICY "imágenes públicas"
  ON storage.objects FOR SELECT
  USING (bucket_id = 'product-images');

-- Solo el artesano dueño puede subir (la carpeta raíz es su user ID)
CREATE POLICY "artesano sube su imagen"
  ON storage.objects FOR INSERT
  WITH CHECK (
    bucket_id = 'product-images'
    AND auth.uid()::text = (storage.foldername(name))[1]
  );

-- Solo el artesano dueño puede borrar
CREATE POLICY "artesano borra su imagen"
  ON storage.objects FOR DELETE
  USING (
    bucket_id = 'product-images'
    AND auth.uid()::text = (storage.foldername(name))[1]
  );
