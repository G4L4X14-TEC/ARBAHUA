Guía Paso a Paso: Reconstrucción Completa de ARBAHUA
FASE 1 — Crear el Proyecto en Supabase
Paso 1. Ve a https://supabase.com e inicia sesión (o regístrate).

Paso 2. Haz clic en "New Project".

Paso 3. Completa el formulario:

Name: ARBAHUA
Database Password: elige una contraseña segura y guárdala (la necesitarás si conectas directamente a Postgres)
Region: South America (São Paulo) — la más cercana a México
Haz clic en "Create new project"
⏳ Espera ~2 minutos mientras Supabase provisiona el proyecto.

Paso 4. Una vez creado, ve a Settings → API y copia estos dos valores:

Project URL → será tu NEXT_PUBLIC_SUPABASE_URL
anon public key → será tu NEXT_PUBLIC_SUPABASE_ANON_KEY
FASE 2 — Configurar Autenticación (Auth)
Paso 5. En el menú lateral ve a Authentication → URL Configuration.

Configura:

Site URL: http://localhost:9002 (para desarrollo)
Redirect URLs: añade http://localhost:9002/update-password
Esto es necesario porque forgot-password/page.tsx redirige a /update-password al resetear contraseña.

Paso 6. Ve a Authentication → Settings (Email) y verifica que:

"Enable email confirmations" esté activado ✅

🔵 Bloque 5 — Trigger automático de perfil al registrarse
SQL
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
  INSERT INTO public.usuarios (id, nombre, email, rol)
  VALUES (
    NEW.id,
    COALESCE(NEW.raw_user_meta_data->>'nombre', ''),
    NEW.email,
    COALESCE((NEW.raw_user_meta_data->>'rol')::public.rol_type, 'cliente')
  )
  ON CONFLICT (id) DO NOTHING;
  RETURN NEW;
END;
$$;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

SQL
-- ===== USUARIOS =====
CREATE POLICY "ver propio perfil" ON public.usuarios
  FOR SELECT USING (auth.uid() = id);
CREATE POLICY "crear propio perfil" ON public.usuarios
  FOR INSERT WITH CHECK (auth.uid() = id);
CREATE POLICY "actualizar propio perfil" ON public.usuarios
  FOR UPDATE USING (auth.uid() = id);

-- ===== TIENDAS =====
CREATE POLICY "tiendas activas son públicas" ON public.tiendas
  FOR SELECT USING (estado = 'activa' OR artesano_id = auth.uid());
CREATE POLICY "artesano crea su tienda" ON public.tiendas
  FOR INSERT WITH CHECK (artesano_id = auth.uid());
CREATE POLICY "artesano edita su tienda" ON public.tiendas
  FOR UPDATE USING (artesano_id = auth.uid());

-- ===== PRODUCTOS =====
CREATE POLICY "productos activos son públicos" ON public.productos
  FOR SELECT USING (
    estado = 'activo' OR
    tienda_id IN (SELECT id FROM public.tiendas WHERE artesano_id = auth.uid())
  );
CREATE POLICY "artesano crea productos" ON public.productos
  FOR INSERT WITH CHECK (
    tienda_id IN (SELECT id FROM public.tiendas WHERE artesano_id = auth.uid())
  );
CREATE POLICY "artesano edita sus productos" ON public.productos
  FOR UPDATE USING (
    tienda_id IN (SELECT id FROM public.tiendas WHERE artesano_id = auth.uid())
  );
CREATE POLICY "artesano elimina sus productos" ON public.productos
  FOR DELETE USING (
    tienda_id IN (SELECT id FROM public.tiendas WHERE artesano_id = auth.uid())
  );

-- ===== CATEGORIAS =====
CREATE POLICY "categorias son públicas" ON public.categorias
  FOR SELECT USING (true);

-- ===== PRODUCTO_CATEGORIA =====
CREATE POLICY "relacion producto-cat es pública" ON public.producto_categoria
  FOR SELECT USING (true);
CREATE POLICY "artesano gestiona categorías de sus productos" ON public.producto_categoria
  FOR ALL USING (
    producto_id IN (
      SELECT p.id FROM public.productos p
      JOIN public.tiendas t ON p.tienda_id = t.id
      WHERE t.artesano_id = auth.uid()
    )
  );

-- ===== IMAGENES_PRODUCTOS =====
CREATE POLICY "imágenes son públicas" ON public.imagenes_productos
  FOR SELECT USING (true);
CREATE POLICY "artesano sube imágenes" ON public.imagenes_productos
  FOR INSERT WITH CHECK (
    producto_id IN (
      SELECT p.id FROM public.productos p
      JOIN public.tiendas t ON p.tienda_id = t.id
      WHERE t.artesano_id = auth.uid()
    )
  );
CREATE POLICY "artesano edita imágenes" ON public.imagenes_productos
  FOR UPDATE USING (
    producto_id IN (
      SELECT p.id FROM public.productos p
      JOIN public.tiendas t ON p.tienda_id = t.id
      WHERE t.artesano_id = auth.uid()
    )
  );
CREATE POLICY "artesano elimina imágenes" ON public.imagenes_productos

    producto_id IN (
      SELECT p.id FROM public.productos p
      JOIN public.tiendas t ON p.tienda_id = t.id
      WHERE t.artesano_id = auth.uid()
    )
  );

-- ===== CARRITOS E ITEMS =====
CREATE POLICY "carrito propio" ON public.carritos
  FOR ALL USING (cliente_id = auth.uid()) WITH CHECK (cliente_id = auth.uid());
CREATE POLICY "items del carrito propio" ON public.items_carrito
  FOR ALL USING (
    carrito_id IN (SELECT id FROM public.carritos WHERE cliente_id = auth.uid())
  );

-- ===== DIRECCIONES =====
CREATE POLICY "direcciones propias" ON public.direcciones
  FOR ALL USING (cliente_id = auth.uid()) WITH CHECK (cliente_id = auth.uid());

-- ===== PEDIDOS, DETALLES Y PAGOS =====
CREATE POLICY "pedidos propios" ON public.pedidos
  FOR ALL USING (cliente_id = auth.uid()) WITH CHECK (cliente_id = auth.uid());
CREATE POLICY "ver detalles de pedidos propios" ON public.detalle_pedido
  FOR SELECT USING (

Paso 11. En Storage → Policies, agrega estas políticas para el bucket:

SQL
-- Cualquiera puede ver las imágenes
CREATE POLICY "imágenes públicas"
  ON storage.objects FOR SELECT
  USING (bucket_id = 'product-images');

-- Solo el artesano dueño puede subir (la carpeta raíz es su user ID)
CREATE POLICY "artesano sube su imagen"
  ON storage.objects FOR INSERT
  WITH CHECK (
    bucket_id = 'product-images'
    AND auth.uid()::text = (storage.foldername(name))[1]
  );

-- Solo el artesano dueño puede borrar
CREATE POLICY "artesano borra su imagen"
  ON storage.objects FOR DELETE
  USING (
    bucket_id = 'product-images'
    AND auth.uid()::text = (storage.foldername(name))[1]
  );

CREATE POLICY "imágenes públicas" ON storage.objects
  FOR SELECT USING (bucket_id = 'product-images');

CREATE POLICY "artesano sube imagen" ON storage.objects
  FOR INSERT WITH CHECK (
    bucket_id = 'product-images'
    AND auth.uid()::text = (storage.foldername(name))[1]
  );

CREATE POLICY "artesano borra imagen" ON storage.objects

  );
